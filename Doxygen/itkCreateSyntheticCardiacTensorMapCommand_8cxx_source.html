00150 TensorType GetProlateTensor (PointType xi, double* range, double l_helix, double l_sheet, double e2)
00151 {
00152   double xmin=range[0];
00153   double xmax=range[1];
00154   double amin=-l_helix * vnl_math::pi / 180;
00155   double amax= l_helix * vnl_math::pi / 180;
00156   
00157   double x = (xi[0] - xmin) / (xmax - xmin);
00158   double a = - (amax-amin) * x - amin;
00159   double b = std::sin ( 5.0 * xi[2] ) * l_sheet * ( 1.0 + std::cos (x * 2.0 * vnl_math::pi) ) * vnl_math::pi / 360;
00160 
00161   if (e2 < 0.21) e2 = 0.21;
00162   if (e2 > 0.99) e2 = 0.99;
00163   
00164   TensorType::MatrixType v, e, r, rota, rotb;
00165   v.Fill (0.0); rota.Fill (0.0); rotb.Fill (0.0);
00166   
00167   v[2][0] = 1; v[0][1] = 1; v[1][2] = 1;
00168   e[0][0] = 1.0; e[1][1] = e2; e[2][2] = 0.2;
00169   
00170   r = v * e * TensorType::MatrixType (v.GetTranspose());
00171   
00172   rota[0][0] =  1.0;
00173   rota[1][1] =   std::cos (-a);
00174   rota[2][1] = - std::sin (-a);
00175   rota[1][2] =   std::sin (-a);
00176   rota[2][2] =   std::cos (-a);  
00177 
00178   rotb[0][0] =   std::cos (b);
00179   rotb[1][0] = - std::sin (b);
00180   rotb[0][1] =   std::sin (b);
00181   rotb[1][1] =   std::cos (b);
00182   rotb[2][2] =   1.0;
00183   
00184   TensorType T;
00185   T.SetVnlMatrix ( r.GetVnlMatrix() );
00186   
00187   T = T.ApplyMatrix (rotb);
